```bash

python "Omega Control/00 Rules/python/omega_run.py"

# Onboard (first time)
python "Omega Control/00 Rules/python/omega_onboard.py"

# Publish to GitHub
python "Omega Control/00 Rules/python/omega_publish.py" "your message"
```
